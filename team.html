function renderStatus() { const now = new Date(), lDate = getLogicDate(now), shift = getShift(lDate), isC = shift.name === 'C'; const hr = now.getHours(), min = now.getMinutes(); let totalMin = hr * 60 + min; if (hr < 7) totalMin += 1440; // 새벽 시간대 전날 근무 연장 처리 // 메시지 처리 let msg = "", isBig = false; if (isC) { if (hr === 6 && min >= 20 && min < 40) msg = "수고하셨습니다. 잘 마무리 하시고
다음 조와 근무 교대를 준비하십시오."; else if (hr === 6 && min >= 40) msg = "고생하셨습니다.
근무교대 시간입니다."; else msg = "🛡️ 현재 C조 근무 시간입니다.
오늘도 안전에 만전을 기해 주십시오!"; } else { isBig = true; if (hr === 7 && min < 30) msg = "✨ 밤새 고생 많으셨습니다.
안전하게 퇴근하십시오!"; else msg = "😴 오늘은 휴무일입니다.
편안한 휴식 되세요."; } document.getElementById('time-msg').innerHTML = `
${msg}
`; // 근무자 및 카드 표출 로직 수정 const nextD = getNextDutyDate(); const currentWs = getWorkers(now) || ["-","-","-","-"]; // 현재 실제 근무자 우선 추출 const cardArea = document.getElementById('worker-cards'); if (isC) { cardArea.classList.remove('hidden'); let curIdx = -1; DATA_LIST.forEach((r, i) => { let [sH, sM] = r[0].split(':').map(Number), [eH, eM] = r[1].split(':').map(Number); let sT = sH * 60 + sM, eT = eH * 60 + eM; if (sH < 7) sT += 1440; if (eH < 7 || (eH === 7 && eM === 0)) eT += 1440; if (totalMin >= sT && totalMin < eT) curIdx = i; }); const curPos = (curIdx !== -1) ? DATA_LIST[curIdx] : ["-","-","휴게","휴게","휴게","휴게"]; cardArea.innerHTML = currentWs.map((w, idx) => `
${['조장','성의','의산A','의산B'][idx]} ${w}
${curPos[idx+2]}
`).join(''); } else { cardArea.classList.add('hidden'); } document.getElementById('next-duty-notice').innerText = nextD ? `다음 근무는 ${nextD.getMonth()+1}월 ${nextD.getDate()}일 입니다.` : ""; // ... 이하 테이블 렌더링 로직 동일 ... } // 다음 근무일 계산 로직 수정 function getNextDutyDate() { let now = new Date(); // 새벽 7시 이전이면 '오늘'도 다음 근무 후보에서 제외하지 않도록 보정 let startCheck = (now.getHours() < 7) ? 0 : 1; for(let i = startCheck; i < 10; i++) { let check = new Date(); check.setDate(now.getDate() + i); if(getShift(getLogicDate(check)).name === 'C') { // 현재 근무 중인 날짜는 건너뜀 if(getShift(getLogicDate(now)).name === 'C' && getLogicDate(check).getTime() === getLogicDate(now).getTime()) continue; return check; } } return null; }
